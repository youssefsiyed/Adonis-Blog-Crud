npm install -g @adonisjs/cli       /// install the client

adonis new app                    /// create project name app

adonis serve --dev                  /// run the server

adonis make:view home                /// create home.html

adonis make:controller User           //// create controller

npm install mysql                     // driver mysql

adonis make:migration users           // users model

adonis migration:run                  // run the migration

adonis make:model Person              // create model person

npm install @adonisjs/validator       /// valider les champs
